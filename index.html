    // Display courts

    function displayCourts() {

      const container = document.getElementById('courtList');

      container.innerHTML = '';

      courts.forEach(court => {

        const card = document.createElement('div');

        card.className = 'court-card';

        card.innerHTML = `

          <h3>${court.name}</h3>

          <p>${court.address}</p>

          <p><strong>Courts:</strong> ${court.courts}</p>

          ${court.link}Book on Venue Site</a>

        `;

        container.appendChild(card);

      });

 

      // Populate booking dropdown

      const select = document.getElementById('courtSelect');

      courts.forEach(court => {

        const option = document.createElement('option');

        option.value = court.name;

        option.textContent = court.name;

        select.appendChild(option);

      });

    }

 

    function filterCourts() {

      const query = document.getElementById('search').value.toLowerCase();

      const container = document.getElementById('courtList');

      container.innerHTML = '';

      courts.filter(court => court.name.toLowerCase().includes(query) || court.address.toLowerCase().includes(query))

        .forEach(court => {

          const card = document.createElement('div');

          card.className = 'court-card';

          card.innerHTML = `

            <h3>${court.name}</h3>

            <p>${court.address}</p>

            <p><strong>Courts:</strong> ${court.courts}</p>

            ${court.link}Book on Venue Site</a>

          `;

          container.appendChild(card);

        });

    }

 

    function initMap() {

      const map = new google.maps.Map(document.getElementById('map'), {

        zoom: 12,

        center: { lat: 51.4545, lng: -2.5879 }

      });

      courts.forEach(court => {

        new google.maps.Marker({

          position: { lat: court.lat, lng: court.lng },

          map: map,

          title: court.name

        });

      });

    }

    document.getElementById('bookingForm').addEventListener('submit', function(e) {

      e.preventDefault();

      const booking = {

        court: document.getElementById('courtSelect').value,

        date: document.getElementById('date').value,

        time: document.getElementById('time').value,

        name: document.getElementById('name').value,

        email: document.getElementById('email').value

      };

 

      // Save to Firestore

      db.collection('bookings').add(booking);

 

      // Send email via EmailJS

      emailjs.send('YOUR_SERVICE_ID', 'YOUR_TEMPLATE_ID', booking)

        });
